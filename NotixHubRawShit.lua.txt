local function loadConfig()
    pcall(function()
        local has = false; pcall(function() has = _isfile(CONFIG_FILE) end)
        if not has then return end
        local raw; pcall(function() raw = _readfile(CONFIG_FILE) end)
        if not raw then return end

        if ok and dec then for k,v in pairs(dec) do Cfg[k] = v end end
    end)
end

local function mkCorner(p, r)
    local c = Instance.new("UICorner", p); c.CornerRadius = UDim.new(0, r or 6); return c
end
local function mkStroke(p, col, th)
    local s = Instance.new("UIStroke", p)
    s.Color = col; s.Thickness = th or 1
    s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border; return s
end

-- Pulsing glow on a stroke
local function addPulse(stroke, col1, col2)
    task.spawn(function()
        local t = 0
        while stroke and stroke.Parent do
            t = t + 0.05
            local alpha = (math.sin(t * 2) + 1) / 2
            stroke.Transparency = 0.1 + alpha * 0.5
            task.wait(0.05)
        end
    end)
end


-- Purple strobe (cycles a UIStroke through the purple palette used by the
-- "Notix Hub User" billboard, so toggle/menu chrome matches the detection tag).
local SILVER_STROBE = {
    Color3.fromRGB(90,  20, 140),
    Color3.fromRGB(140, 50, 200),
    Color3.fromRGB(190, 100, 255),
    Color3.fromRGB(220, 160, 255),
    Color3.fromRGB(140, 50, 200),
}
local function addSilverStrobe(stroke)
    task.spawn(function()
        local i = 1
        while stroke and stroke.Parent do
            local nxt = SILVER_STROBE[(i % #SILVER_STROBE) + 1]
            pcall(function()
                TweenService:Create(stroke, TweenInfo.new(0.45, Enum.EasingStyle.Sine), {Color = nxt}):Play()
            end)
            i = i + 1
            task.wait(0.45)
        end
    end)
end

-- ============================================================
-- TAUNT LOGIC (sends "Notix hub" every 0.5s)
-- ============================================================
local function sendChat(msg)
    -- Try new TextChatService first
    local success = pcall(function()
        local generalChannel = TextChatService.TextChannels:FindFirstChild("RBXGeneral")
        if generalChannel then
            generalChannel:SendAsync(msg)
        end
    end)
    -- Fallback to legacy
    if not success then
        pcall(function()
            local rs = game:GetService("ReplicatedStorage")
            local ev = rs:FindFirstChild("DefaultChatSystemChatEvents")
            if ev then
                local smr = ev:FindFirstChild("SayMessageRequest")
                if smr then smr:FireServer(msg, "All") end
            end
        end)
    end
end

local function stopTaunt(btn)
    State.tauntActive = false
    if State._tauntThread then
        pcall(function() task.cancel(State._tauntThread) end)
        State._tauntThread = nil
    end
    if btn then btn.setActive(false); btn.sub.Text = "OFF" end
end

local function startTaunt(btn)
    if State.tauntActive then stopTaunt(btn); return end
    State.tauntActive = true
    if btn then btn.setActive(true); btn.sub.Text = "ON" end
    State._tauntThread = task.spawn(function()
        while State.tauntActive do
            pcall(function() sendChat("Notix on Top 🥤🐬") end)
            task.wait(0.5)
        end
    end)
end

-- Build one button at (col, row) zero-indexed
local function makeEnvyBtn(label, col, row, onClick)
    local x = MB_PAD + col * (MB_W + MB_PAD)
    local y = MB_PAD + row * (MB_H + MB_PAD)
    local btn = Instance.new("TextButton", mbPanel)
    btn.Size             = UDim2.new(0, MB_W, 0, MB_H)
    btn.Position         = UDim2.new(0, x, 0, y)
    btn.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    btn.BorderSizePixel  = 0
    btn.Text             = label
    btn.TextColor3       = Color3.fromRGB(255, 255, 255)
    btn.Font             = Enum.Font.GothamBlack
    btn.TextSize         = 11
    btn.TextWrapped      = true
    btn.ZIndex           = 22
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, MB_CORNER)
    btn.AutoButtonColor = false
    addSilverStrobe(mkStroke(btn, Color3.fromRGB(160, 70, 220), 1))
    local isOn = false
    local function setActive(on)
        isOn = on
        TweenService:Create(btn, TweenInfo.new(0.15), {
            BackgroundColor3 = on and Color3.fromRGB(45, 45, 45) or Color3.fromRGB(0, 0, 0),
        }):Play()
    end
    if onClick then
        btn.MouseButton1Click:Connect(function() pcall(onClick) end)
    end
    local sub = Instance.new("TextLabel", btn)
    sub.Size = UDim2.new(0,0,0,0); sub.BackgroundTransparency = 1
    sub.Text = ""; sub.TextColor3 = Color3.fromRGB(0,0,0)
    sub.Font = Enum.Font.Gotham; sub.TextSize = 1; sub.ZIndex = 1
    local dot = Instance.new("Frame", btn)
    dot.Size = UDim2.new(0,0,0,0); dot.BackgroundTransparency = 1; dot.BorderSizePixel = 0; dot.ZIndex = 1
    local kc = _stubKeyChip(btn)
    return {frame=btn, clk=btn, lbl=btn, sub=sub, dot=dot, keyChip=kc, setActive=setActive}
end

local _editorOpen = false
local SNAP = 5  -- snap to 5px grid

local function openPanelEditor()
    if _editorOpen then return end
    _editorOpen = true

    local ALL_BTNS = {"tauntBtn","drop","autoL","autoR","aimbot","tpDown","float_","floatUp"}
    local defaults = _defaultBtnOffsets()

    -- Use constants — AbsoluteSize may not be ready yet
    local epW = MB_COLS * MB_W + (MB_COLS + 1) * MB_PAD
    local epH = MB_ROWS * MB_H + (MB_ROWS + 1) * MB_PAD

    overlay.Size            = UDim2.new(1, 0, 1, 0)

    overlay.BackgroundColor3= Color3.fromRGB(0, 0, 0)

    overlay.BorderSizePixel = 0
    overlay.ZIndex          = 100
    overlay.Active          = true

    -- ── TITLE (above the panel, outside it so ClipsDescendants won't hide it) ──
    local titleLbl = Instance.new("TextLabel", overlay)
    titleLbl.Size            = UDim2.new(0, epW + 40, 0, 22)
    titleLbl.Position        = UDim2.new(0.5, -math.floor((epW+40)/2), 0.5, -math.floor(epH/2) - 30)
    titleLbl.BackgroundTransparency = 1
    titleLbl.Text            = "DRAG TO REPOSITION  •  SNAPS TO GRID"
    titleLbl.TextColor3      = Color3.fromRGB(180, 180, 180)
    titleLbl.Font            = Enum.Font.GothamBold
    titleLbl.TextSize        = 11
    titleLbl.TextXAlignment  = Enum.TextXAlignment.Center
    titleLbl.ZIndex          = 105

    -- ── EDITOR PANEL ─────────────────────────────────────────
    local edPanel = Instance.new("Frame", overlay)
    edPanel.Name             = "EditorPanel"
    edPanel.Size             = UDim2.new(0, epW, 0, epH)
    edPanel.Position         = UDim2.new(0.5, -math.floor(epW/2), 0.5, -math.floor(epH/2))
    edPanel.BackgroundColor3 = Color3.fromRGB(8, 8, 8)
    edPanel.BorderSizePixel  = 0
    edPanel.ZIndex           = 101
    edPanel.ClipsDescendants = true
    Instance.new("UICorner", edPanel).CornerRadius = UDim.new(0, MB_CORNER + 4)
    addSilverStrobe(mkStroke(edPanel, Color3.fromRGB(160, 70, 220), 2))

    -- Faint grid dots so snapping is visible
    for col = 0, math.floor(epW / SNAP) do
        for row = 0, math.floor(epH / SNAP) do
            local dot = Instance.new("Frame", edPanel)
            dot.Size = UDim2.new(0, 1, 0, 1)
            dot.Position = UDim2.new(0, col * SNAP, 0, row * SNAP)
            dot.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
            dot.BackgroundTransparency = 0
            dot.BorderSizePixel = 0
            dot.ZIndex = 101
        end
    end

    -- ── SHARED DRAG STATE ────────────────────────────────────
    -- Only one ghost dragged at a time; all share a single UIS.InputChanged
    local activeDrag = nil  -- {ghost, draftName, startInput, startOff, allGhosts}

    -- ── BUILD GHOST BUTTONS ──────────────────────────────────
    local draftPos  = {}  -- name -> {xOff, yOff}
    local ghostMap  = {}  -- name -> ghost frame
    local allGhosts = {}

    local function snapTo(v) return math.floor(v / SNAP + 0.5) * SNAP end

    for _, name in ipairs(ALL_BTNS) do
        local b = bigBtns[name]
        if b and b.frame then
            local orig = b.frame
            -- Start from saved position if available, else default
            local saved = Cfg.btnPositions[name]
            local xOff = saved and saved[1] or defaults[name][1]
            local yOff = saved and saved[2] or defaults[name][2]
            draftPos[name] = {xOff, yOff}

            local ghost = Instance.new("TextButton", edPanel)
            ghost.Name               = "ghost_" .. name
            ghost.Size               = UDim2.new(0, MB_W, 0, MB_H)
            ghost.Position           = UDim2.new(0, xOff, 0, yOff)
            ghost.BackgroundColor3   = Color3.fromRGB(12, 12, 12)
            ghost.BackgroundTransparency = 0
            ghost.Text               = orig.Text
            ghost.TextColor3         = Color3.fromRGB(255, 255, 255)
            ghost.Font               = Enum.Font.GothamBlack
            ghost.TextSize           = 11
            ghost.TextWrapped        = true
            ghost.BorderSizePixel    = 0
            ghost.ZIndex             = 103
            ghost.AutoButtonColor    = false
            Instance.new("UICorner", ghost).CornerRadius = UDim.new(0, MB_CORNER)
            mkStroke(ghost, Color3.fromRGB(120, 120, 130), 1)

            ghostMap[name] = ghost
            table.insert(allGhosts, ghost)

            local capturedName = name
            ghost.InputBegan:Connect(function(inp)
                if inp.UserInputType ~= Enum.UserInputType.MouseButton1
                and inp.UserInputType ~= Enum.UserInputType.Touch then return end
                if activeDrag then return end  -- another button already dragging
                activeDrag = {
                    ghost     = ghost,
                    name      = capturedName,
                    startPos  = inp.Position,
                    startOff  = {draftPos[capturedName][1], draftPos[capturedName][2]},
                }
                -- Highlight active ghost
                TweenService:Create(ghost, TweenInfo.new(0.06), {
                    BackgroundColor3 = Color3.fromRGB(30, 30, 30),
                }):Play()
                inp.Changed:Connect(function()
                    if inp.UserInputState == Enum.UserInputState.End then
                        if activeDrag and activeDrag.ghost == ghost then
                            -- Snap on release
                            local sx = snapTo(draftPos[capturedName][1])
                            local sy = snapTo(draftPos[capturedName][2])
                            sx = math.clamp(sx, 0, epW - MB_W)
                            sy = math.clamp(sy, 0, epH - MB_H)
                            draftPos[capturedName] = {sx, sy}
                            ghost.Position = UDim2.new(0, sx, 0, sy)
                            TweenService:Create(ghost, TweenInfo.new(0.06), {
                                BackgroundColor3 = Color3.fromRGB(12, 12, 12),
                            }):Play()
                            activeDrag = nil
                        end
                    end

local _floatActive = false
local _floatThread = nil
local _floatVF, _floatAtt = nil, nil

local function stopFloat()
    _floatActive = false
    if _floatThread then pcall(function() task.cancel(_floatThread) end); _floatThread = nil end
    pcall(function()
        if _floatVF then _floatVF:Destroy(); _floatVF = nil end
        if _floatAtt then _floatAtt:Destroy(); _floatAtt = nil end
        local c = LP.Character; if not c then return end
        local root = c:FindFirstChild("HumanoidRootPart"); if not root then return end
        local bp = root:FindFirstChild("JHubFloatBP"); if bp then bp:Destroy() end
    end)
    if bigBtns.floatUp then
        bigBtns.floatUp.setActive(false)
        bigBtns.floatUp.sub.Text = "OFF"
    end
end

local function startFloat()
    local c = LP.Character; if not c then return end
    local root = c:FindFirstChild("HumanoidRootPart"); if not root then return end
    local hum = c:FindFirstChildOfClass("Humanoid")
    local fh = State.floatHeight
    local targetY = root.Position.Y + fh

    -- Clean any prior float artifacts
    local oldBP = root:FindFirstChild("JHubFloatBP"); if oldBP then oldBP:Destroy() end
    if _floatVF then _floatVF:Destroy() end
    if _floatAtt then _floatAtt:Destroy() end

    -- VectorForce that cancels gravity + adds tiny corrective lift.
    -- This does NOT change humanoid state (no Freefall/Running mutation),
    -- so the equipped tool is never dropped.
    _floatAtt = Instance.new("Attachment")
    _floatAtt.Name = "JHubFloatAtt"
    _floatAtt.Parent = root
    _floatVF = Instance.new("VectorForce")
    _floatVF.Name = "JHubFloatVF"
    _floatVF.Attachment0 = _floatAtt
    _floatVF.RelativeTo = Enum.ActuatorRelativeTo.World
    _floatVF.ApplyAtCenterOfMass = true
    _floatVF.Force = Vector3.new(0, 0, 0)
    _floatVF.Parent = root

    _floatActive = true
    if bigBtns.floatUp then
        bigBtns.floatUp.setActive(true)
        bigBtns.floatUp.sub.Text = "ON " .. fh .. "st"
    end

    _floatThread = task.spawn(function()
        while _floatActive do
            RunService.Heartbeat:Wait()
            local cc = LP.Character; if not cc then stopFloat(); break end
            local rt = cc:FindFirstChild("HumanoidRootPart")
            if not rt then stopFloat(); break end
            -- Estimate assembly mass
            local mass = rt.AssemblyMass
            if not mass or mass <= 0 then mass = 14 end
            local g = workspace.Gravity
            local dy = targetY - rt.Position.Y
            local vy = rt.AssemblyLinearVelocity.Y
            -- PD-ish corrective force on Y, cancels gravity
            local corrective = (dy * 60) - (vy * 8)
            _floatVF.Force = Vector3.new(0, mass * (g + corrective), 0)
        end
    end)
end

-- Exact opium message_11 runDropBrainrot logic, momentary (auto-offs every press)
local function runDropBrainrot()
    if State.dropEnabled then return end
    State.dropEnabled = true
    bigBtns.drop.setActive(true)
    bigBtns.drop.sub.Text = "ON"
    task.spawn(function()
        local colConn = RunService.Stepped:Connect(function()
            if not State.dropEnabled then return end

                if p ~= LP and p.Character then
                    for _, part in ipairs(p.Character:GetChildren()) do
                        if part:IsA("BasePart") then pcall(function() part.CanCollide = false end) end
                    end
                end

            end

        bigBtns.drop.sub.Text = "OFF"
    end)

local function _activateAutoBtn(side)
    local activeBtn  = (side == "L") and bigBtns.autoL or bigBtns.autoR
    local inactiveBtn = (side == "L") and bigBtns.autoR or bigBtns.autoL

local function startAutoPlay(side)
    -- If already running the same side, stop it
    if State.autoEnabled and _autoSide == side then stopAutoPlay(); return end
    -- If running the other side, stop first
    if State.autoEnabled then stopAutoPlay() end

    _autoSide = side
    State.autoEnabled = true
    _autoPhase = 1
    _autoDelaying = false
    if State.aimbotActive then stopAimbot() end

    _activateAutoBtn(side)

    -- Select position tables for this side
    local POS1, POS_END, POS_RET, POS4, POS_FIN
    if side == "L" then
        POS1    = L_POS_1
        POS_END = L_POS_END
        POS_RET = L_POS_RETURN
        POS4    = L_POS_4
        POS_FIN = L_POS_FINAL
    else
        POS1    = R_POS_1
        POS_END = R_POS_END
        POS_RET = R_POS_RETURN
        POS4    = R_POS_4
        POS_FIN = R_POS_FINAL
    end

    _autoPlayConn = RunService.Heartbeat:Connect(function()
        if not State.autoEnabled then stopAutoPlay(); return end
        if State.countdownActive then return end
        local c = LP.Character; if not c then return end
        local root = c:FindFirstChild("HumanoidRootPart")
        local hum2 = c:FindFirstChildOfClass("Humanoid")
        if not root or not hum2 then return end
        if hum2.Health <= 0 then stopAutoPlay(); return end

        local function moveTo(target, spd)
            local d = (target - root.Position)
            local mv = Vector3.new(d.X, 0, d.Z).Unit
            hum2:Move(mv, false)
            root.AssemblyLinearVelocity = Vector3.new(mv.X * spd, root.AssemblyLinearVelocity.Y, mv.Z * spd)
        end

        local function reachedXZ(target, threshold)
            local tgt = Vector3.new(target.X, root.Position.Y, target.Z)
            return (tgt - root.Position).Magnitude < threshold
        end

        if _autoPhase == 1 then
            -- Go: start → POS1
            if reachedXZ(POS1, 1.5) then
                _autoPhase = 2
                moveTo(POS_END, 58)
                return
            end
            moveTo(POS1, 58)

        elseif _autoPhase == 2 then
            -- Go: POS1 → POS_END
            if reachedXZ(POS_END, 1.5) then
                if not _autoDelaying then
                    _autoDelaying = true
                    hum2:Move(Vector3.zero, false)
                    root.AssemblyLinearVelocity = Vector3.zero
                    local delay = State.autoPlayDelay or 0
                    if delay > 0 then
                        task.delay(delay, function()
                            if State.autoEnabled then
                                _autoDelaying = false
                                _autoPhase = 3
                            end
                        end)
                    else
                        _autoDelaying = false
                        _autoPhase = 3
                    end
                end
                return
            end
            moveTo(POS_END, 58)

        elseif _autoPhase == 3 then
            -- Return: POS_END → POS_RETURN (back to start area)
            if reachedXZ(POS_RET, 2) then
                _autoPhase = 4
                moveTo(POS4, 29)
                return
            end
            moveTo(POS_RET, 29)

        elseif _autoPhase == 4 then
            -- Return: POS_RETURN → POS_4
            if reachedXZ(POS4, 2) then
                _autoPhase = 5
                moveTo(POS_FIN, 29)
                return
            end
            moveTo(POS4, 29)

        elseif _autoPhase == 5 then
            -- Return: POS_4 → POS_FINAL  (end of route)
            if reachedXZ(POS_FIN, 2) then
                hum2:Move(Vector3.zero, false)
                root.AssemblyLinearVelocity = Vector3.zero
                stopAutoPlay()
                return
            end
            moveTo(POS_FIN, 29)
        end
    end)
end

-- EMA prediction state (best fit for noisy human-driven targets:
-- exponential moving average smooths jitter while staying responsive,

-- ── helpers ──────────────────────────────────────────────────
local function findAnyTool()
    local c = LP.Character
    if c then for _, v in ipairs(c:GetChildren()) do if v:IsA("Tool") then return v end end end
    local bp = LP:FindFirstChildOfClass("Backpack")
    if bp then for _, v in ipairs(bp:GetChildren()) do if v:IsA("Tool") then return v end end end
    return nil
end

local function pushHistory(tbl, key, val, maxLen)
    if not tbl[key] then tbl[key] = {} end
    table.insert(tbl[key], val)
    if #tbl[key] > maxLen then table.remove(tbl[key], 1) end
end

local function avgHistory(tbl, key)
    local hist = tbl[key]
    if not hist or #hist == 0 then return Vector3.zero end
    local sum = Vector3.zero
    for _, v in ipairs(hist) do sum = sum + v end
    return sum / #hist
end

-- Derive an angular-velocity-based arc offset:
-- If the target is spinning/strafing, their centre of rotation is offset from their
-- HumanoidRootPart. We approximate the tangential velocity contribution from angular
-- motion using ω × r, where r = vector from their root to the predicted linear position.
local function angularContrib(targetRoot, avgAngVel, predLinearPos)
    local r = predLinearPos - targetRoot.Position
    -- ω × r gives the tangential velocity at distance r from the rotation axis
    return avgAngVel:Cross(r)
end

local function getBestTarget()
    if not hrp then return nil, math.huge end
    local best, bestDist = nil, math.huge
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LP and p.Character then
            local tr = p.Character:FindFirstChild("HumanoidRootPart")
            local ph = p.Character:FindFirstChildOfClass("Humanoid")
            if tr and ph and ph.Health > 0 then
                local d = (hrp.Position - tr.Position).Magnitude
                if d < bestDist and d < AIMBOT_RANGE then bestDist = d; best = p end
            end
        end
    end
    return best, bestDist
end

end

    _predVisLabel.TextXAlignment         = Enum.TextXAlignment.Center
    _predVisLabel.TextWrapped            = true
    _predVisLabel.Parent                 = _predVisBB
end

local function _updatePredVis(predPos, info)
    if not _predVisEnabled then return end
    if not _predVisPart then _buildPredVis() end
    pcall(function()
        _predVisPart.CFrame = CFrame.new(predPos)
        -- Format a 3-line readout: predicted position, linear contribution, angular contribution
        local linM = math.floor(info.linContrib.Magnitude * 10 + 0.5) / 10
        local angM = math.floor(info.angContrib.Magnitude * 10 + 0.5) / 10
        local tMs  = math.floor(info.t * 1000 + 0.5)
        _predVisLabel.Text = string.format(
            "PRED  t=%dms\nLin ×%.1f  Ang ×%.1f\n%.1f, %.1f, %.1f",
            tMs, linM, angM,
            predPos.X, predPos.Y, predPos.Z
        )
    end)
end

local function _hidePredVis()
    if _predVisPart then
        pcall(function() _predVisPart.CFrame = CFrame.new(0, -9999, 0) end)
    end
    if _predVisLabel then _predVisLabel.Text = "" end
end

stopAimbot = function()
    State.aimbotActive = false
    if _aimbotConn  then _aimbotConn:Disconnect();  _aimbotConn  = nil end
    if _aimbotEquip then task.cancel(_aimbotEquip); _aimbotEquip = nil end
    pcall(function()
        local c = LP.Character; if not c then return end
        local root = c:FindFirstChild("HumanoidRootPart")
        if root then root.AssemblyLinearVelocity = Vector3.zero end
        local hum2 = c:FindFirstChildOfClass("Humanoid")
        if hum2 then hum2.PlatformStand = false end
    end)
    _swingCD = false
    _linHistory = {}; _angHistory = {}; _lastPos = {}
    _emaVel = {}; _emaAcc = {}; _prevVel = {}; _prevTick = {}
    _hidePredVis()
    bigBtns.aimbot.setActive(false)
    bigBtns.aimbot.sub.Text = "OFF"
end

startAimbot = function()
    State.aimbotActive = true
    if State.autoEnabled then stopAutoPlay() end
    bigBtns.aimbot.setActive(true)
    bigBtns.aimbot.sub.Text = "ON"

    _aimbotConn = RunService.Heartbeat:Connect(function(dt)
        if not State.aimbotActive then stopAimbot(); return end
        local c = LP.Character; if not c then return end
        local root = c:FindFirstChild("HumanoidRootPart"); if not root then return end
        local hum2 = c:FindFirstChildOfClass("Humanoid"); if not hum2 then return end

        local target, dist = getBestTarget()

        if target and target.Character then
            local tr = target.Character:FindFirstChild("HumanoidRootPart")
            if tr then
                -- ── EMA velocity + acceleration sampling ──────────────
                local now    = tick()
                local rawVel = tr.AssemblyLinearVelocity
                local pv     = _prevVel[target] or rawVel
                local pt     = _prevTick[target] or now
                local sdt    = math.max(now - pt, 1/240)
                -- raw acceleration from last sample
                local rawAcc = (rawVel - pv) / sdt

                -- EMA update (clamped alphas guard against dt spikes)
                local av = EMA_ALPHA_VEL
                local aa = EMA_ALPHA_ACC
                _emaVel[target] = (_emaVel[target] or rawVel) * (1 - av) + rawVel * av
                _emaAcc[target] = (_emaAcc[target] or rawAcc) * (1 - aa) + rawAcc * aa
                _prevVel[target]  = rawVel
                _prevTick[target] = now

                -- ── prediction time ───────────────────────────────────
                local toTarget    = tr.Position - root.Position
                local timeToReach = math.max(toTarget.Magnitude / AIMBOT_SPEED, 0.04)
                local predTime    = math.min(timeToReach * 0.72, 0.40) * (PRED_SCALE or 1.0)

                -- ── kinematic prediction: pos + v*t + 0.5*a*t^2 ───────
                local emaV   = _emaVel[target]
                local emaA   = _emaAcc[target]
                -- clamp acceleration magnitude to avoid runaway over long predTime
                if emaA.Magnitude > 80 then emaA = emaA.Unit * 80 end
                local linPred = tr.Position + emaV * predTime + emaA * (0.5 * predTime * predTime)

                -- Small upward bias targets upper torso, not the feet
                local aimPos = Vector3.new(linPred.X, linPred.Y + 0.8, linPred.Z)

                _lastPredInfo = {
                    pos        = aimPos,
                    linContrib = emaV * predTime,
                    angContrib = emaA * (0.5 * predTime * predTime),
                    t          = predTime,
                }
                _updatePredVis(aimPos, _lastPredInfo)

                -- ── steer toward predicted point ──────────────────────
                local dir = (aimPos - root.Position).Unit
                hum2.PlatformStand = true
                root.AssemblyLinearVelocity = dir * AIMBOT_SPEED

                if dist <= HIT_DIST then trySwing() end
            end
        else
            root.AssemblyLinearVelocity = Vector3.zero
            hum2.PlatformStand = false
            _hidePredVis()
        end
    end)

    -- Keep tool equipped while aimbot runs
    _aimbotEquip = task.spawn(function()
        while State.aimbotActive do
            pcall(function()
                local c2 = LP.Character; if not c2 then return end
                local hum2 = c2:FindFirstChildOfClass("Humanoid"); if not hum2 then return end
                local tool = findAnyTool()
                if tool and tool.Parent == LP:FindFirstChildOfClass("Backpack") then
                    hum2:EquipTool(tool)
                end
            end)
            task.wait(0.12)
        end
    end)
end

end)

-- ====================== ANTI RAGDOLL (ported from Notix Hub Duels v48) ======================
-- Full velocity-clamped, constraint-cleaning, control-restoring anti-ragdoll.
local function startAntiRag()
    if #AntiRagdollConns > 0 then return end
    local c = LP.Character or LP.CharacterAdded:Wait()
    local humanoid = c:WaitForChild("Humanoid")
    local root     = c:WaitForChild("HumanoidRootPart")
    local animator = humanoid:WaitForChild("Animator")

    local maxVelocity  = 40
    local clampVelocity = 25
    local maxClamp     = 15
    local lastVelocity = Vector3.new(0, 0, 0)

    local function IsRagdollState()
        local state = humanoid:GetState()
        return state == Enum.HumanoidStateType.Physics
            or state == Enum.HumanoidStateType.Ragdoll
            or state == Enum.HumanoidStateType.FallingDown
            or state == Enum.HumanoidStateType.GettingUp
    end

    local function CleanRagdollEffects()
        for _, obj in pairs(c:GetDescendants()) do
            -- Skip our own movers (float / aimbot use these)
            local n = obj.Name or ""
            if n:sub(1,4) == "JHub" then continue end
            local p = obj.Parent
            if p and (p.Name or ""):sub(1,4) == "JHub" then continue end

            if obj:IsA("BallSocketConstraint") or obj:IsA("NoCollisionConstraint")
                or obj:IsA("HingeConstraint")
                or (obj:IsA("Attachment") and (obj.Name == "A" or obj.Name == "B")) then
                pcall(function() obj:Destroy() end)
            elseif obj:IsA("BodyVelocity") or obj:IsA("BodyPosition") or obj:IsA("BodyGyro") then
                pcall(function() obj:Destroy() end)
            elseif obj:IsA("Motor6D") then
                obj.Enabled = true
            end
        end
        -- Stop ragdoll/fall/hurt animations
        for _, track in pairs(animator:GetPlayingAnimationTracks()) do
            local animName = track.Animation and track.Animation.Name:lower() or ""
            if animName:find("rag") or animName:find("fall") or animName:find("hurt") or animName:find("down") then
                track:Stop(0)
            end
        end
    end

    -- Force the character completely upright and standing
    local function ForceStand()
        pcall(function()
            -- Kill floppy flags first
            humanoid.PlatformStand = false
            humanoid.Sit           = false
            humanoid.AutoRotate    = true
            -- Snap root upright: preserve XZ position and Y yaw, zero out pitch/roll
            local pos = root.Position
            local _, yaw, _ = root.CFrame:ToEulerAnglesYXZ()
            root.CFrame = CFrame.new(pos) * CFrame.Angles(0, yaw, 0)
            -- Kill spin / angular momentum
            root.AssemblyAngularVelocity = Vector3.zero
            -- Clamp lateral knockback velocity (preserve Y for gravity)
            local vel = root.AssemblyLinearVelocity
            if vel.Magnitude > maxClamp then
                root.AssemblyLinearVelocity = Vector3.new(
                    vel.X / vel.Magnitude * maxClamp,
                    vel.Y,
                    vel.Z / vel.Magnitude * maxClamp
                )
            end
            -- Transition to running so animations re-attach
            humanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
            task.defer(function()
                pcall(function() humanoid:ChangeState(Enum.HumanoidStateType.Running) end)
            end)
            workspace.CurrentCamera.CameraSubject = humanoid
        end)
    end

    local function ReEnableControls()
        pcall(function()
            require(LP:WaitForChild("PlayerScripts"):WaitForChild("PlayerModule")):GetControls():Enable()
        end)
    end

    -- State-change: snap back instantly the moment a ragdoll state fires
    table.insert(AntiRagdollConns, humanoid.StateChanged:Connect(function()
        if not State.antiRagdollEnabled then return end
        if IsRagdollState() then
            CleanRagdollEffects()
            ForceStand()
            ReEnableControls()
        end
    end))

    -- PlatformStand property watch: some games set this directly without a state change
    table.insert(AntiRagdollConns, humanoid:GetPropertyChangedSignal("PlatformStand"):Connect(function()
        if not State.antiRagdollEnabled then return end
        if humanoid.PlatformStand then
            humanoid.PlatformStand = false
            ForceStand()
            ReEnableControls()
        end
    end))

    -- Heartbeat: continuous enforcement every frame while ragdolled
    table.insert(AntiRagdollConns, RunService.Heartbeat:Connect(function()
        if not State.antiRagdollEnabled then return end
        -- Always kill PlatformStand / Sit even outside ragdoll state
        if humanoid.PlatformStand then humanoid.PlatformStand = false end
        if humanoid.Sit           then humanoid.Sit           = false end
        if not humanoid.AutoRotate then humanoid.AutoRotate   = true  end

        if IsRagdollState() then
            CleanRagdollEffects()
            ForceStand()
            -- Velocity clamp (track delta to catch sudden spikes)
            local vel = root.AssemblyLinearVelocity
            if (vel - lastVelocity).Magnitude > maxVelocity and vel.Magnitude > clampVelocity then
                root.AssemblyLinearVelocity = vel.Unit * math.min(vel.Magnitude, maxClamp)
            end
            lastVelocity = vel
        end
    end))

    -- DescendantAdded: nuke ragdoll constraints the moment they appear
    table.insert(AntiRagdollConns, c.DescendantAdded:Connect(function(obj)
        if not State.antiRagdollEnabled then return end
        -- Immediately kill it then check if we need to stand
        local n = (obj.Name or ""):sub(1,4)
        if n == "JHub" then return end
        if obj:IsA("BallSocketConstraint") or obj:IsA("NoCollisionConstraint")
            or obj:IsA("HingeConstraint") then
            pcall(function() obj:Destroy() end)
            if IsRagdollState() then ForceStand() end
        end
    end))

    -- CharacterAdded: re-bind to the new character
    table.insert(AntiRagdollConns, LP.CharacterAdded:Connect(function(newChar)
        c         = newChar
        humanoid  = newChar:WaitForChild("Humanoid")
        root      = newChar:WaitForChild("HumanoidRootPart")
        animator  = humanoid:WaitForChild("Animator")
        lastVelocity = Vector3.new(0, 0, 0)
        ReEnableControls()
        CleanRagdollEffects()
    end))

    ReEnableControls()
    CleanRagdollEffects()
end

local function stopAntiRag()
    for _, conn in pairs(AntiRagdollConns) do
        pcall(function() conn:Disconnect() end)
    end
    AntiRagdollConns = {}
end

-- Medusa Counter
local _medusaConns = {}
local MEDUSA_CD = 25; local _medusaLastUsed = 0; local _medusaDB = false
local function useMedusa()
    if _medusaDB then return end
    if tick() - _medusaLastUsed < MEDUSA_CD then return end
    local c = LP.Character; if not c then return end
    _medusaDB = true
    local med
    for _, t in ipairs(c:GetChildren()) do
        if t:IsA("Tool") then local n = t.Name:lower(); if n:find("medusa") or n:find("head") or n:find("stone") then med = t; break end end
    end
    if not med then
        local bp = LP:FindFirstChildOfClass("Backpack")
        if bp then for _, t in ipairs(bp:GetChildren()) do
            if t:IsA("Tool") then local n = t.Name:lower(); if n:find("medusa") or n:find("head") or n:find("stone") then med = t; break end end
        end end
    end
    if not med then _medusaDB = false; return end
    if med.Parent ~= c then local hum2 = c:FindFirstChildOfClass("Humanoid"); if hum2 then hum2:EquipTool(med) end end
    pcall(function() med:Activate() end)
    _medusaLastUsed = tick(); _medusaDB = false
end
local function setupMedusa(char)
    for _, c2 in pairs(_medusaConns) do pcall(function() c2:Disconnect() end) end
    _medusaConns = {}
    if not char then return end
    local hum2 = char:FindFirstChildOfClass("Humanoid")
    if hum2 then
        local lastH = hum2.Health
        table.insert(_medusaConns, hum2.HealthChanged:Connect(function(h)
            if not State.medusaCounterEnabled then return end
            if h < lastH - 1 then useMedusa() end
            lastH = h
        end))
        table.insert(_medusaConns, hum2.StateChanged:Connect(function(_, new)
            if not State.medusaCounterEnabled then return end
            if new == Enum.HumanoidStateType.Ragdoll
                or new == Enum.HumanoidStateType.Physics
                or new == Enum.HumanoidStateType.PlatformStanding
                or new == Enum.HumanoidStateType.FallingDown then
                useMedusa()
            end
        end))
    end
    table.insert(_medusaConns, char.DescendantAdded:Connect(function(obj)
        if not State.medusaCounterEnabled then return end
        if obj:IsA("BallSocketConstraint") then useMedusa() end
    end))
    for _, part in ipairs(char:GetDescendants()) do
        if part:IsA("BasePart") then
            table.insert(_medusaConns, part:GetPropertyChangedSignal("Anchored"):Connect(function()
                if not State.medusaCounterEnabled then return end
                if part.Anchored and part.Transparency == 1 then useMedusa() end
            end))
        end
    end
end

-- Bat Counter
local BAT_NAMES = {"Bat","Slap","Iron Slap","Gold Slap","Diamond Slap","Emerald Slap","Ruby Slap","Dark Matter Slap","Flame Slap","Nuclear Slap","Galaxy Slap","Glitched Slap"}
local function findBat()
    local c = LP.Character; if not c then return nil end
    local bp = LP:FindFirstChildOfClass("Backpack")
    for _, nm in ipairs(BAT_NAMES) do
        local t = c:FindFirstChild(nm) or (bp and bp:FindFirstChild(nm)); if t then return t end
    end
    for _, ch in ipairs(c:GetChildren()) do if ch:IsA("Tool") and ch.Name:lower():find("bat") then return ch end end
    if bp then for _, ch in ipairs(bp:GetChildren()) do if ch:IsA("Tool") and ch.Name:lower():find("bat") then return ch end end end
    return nil
end
local function swingBat(bat, char)
    local hum2 = char:FindFirstChildOfClass("Humanoid")
    if bat.Parent ~= char then if hum2 then pcall(function() hum2:EquipTool(bat) end) end; task.wait(0.05) end
    local remote = bat:FindFirstChildOfClass("RemoteEvent")
    if remote then pcall(function() remote:FireServer() end); task.wait(0.15); pcall(function() remote:FireServer() end)
    else pcall(function() bat:Activate() end); task.wait(0.15); pcall(function() bat:Activate() end) end
end
local _batCounterConns = {}

    if not State.batCounterEnabled then return end
    if State.batCounterDebounce then return end
    local char = LP.Character; if not char then return end
    State.batCounterDebounce = true
    task.spawn(function()
        local bat = findBat(); if bat then swingBat(bat, char) end
        task.wait(0.5); State.batCounterDebounce = false
    end)
end
local function _bindBatCounterChar(char)
    for _, c in pairs(_batCounterConns) do pcall(function() c:Disconnect() end) end
    _batCounterConns = {}
    if not char then return end
    local hum2 = char:FindFirstChildOfClass("Humanoid")
    if hum2 then
        local lastH = hum2.Health
        table.insert(_batCounterConns, hum2.HealthChanged:Connect(function(h)
            if h < lastH - 1 then _fireBatCounter() end
            lastH = h
        end))
        table.insert(_batCounterConns, hum2.StateChanged:Connect(function(_, new)
            if new == Enum.HumanoidStateType.Ragdoll
                or new == Enum.HumanoidStateType.Physics
                or new == Enum.HumanoidStateType.PlatformStanding
                or new == Enum.HumanoidStateType.FallingDown then
                _fireBatCounter()
            end
        end))
    end
    table.insert(_batCounterConns, char.DescendantAdded:Connect(function(obj)
        if obj:IsA("BallSocketConstraint") then _fireBatCounter() end
    end))
end
local function startBatCounter()
    if Conns.batCounter then return end
    _bindBatCounterChar(LP.Character)

        task.wait(0.2); _bindBatCounterChar(ch)
    end)
end

    for _, c in pairs(_batCounterConns) do pcall(function() c:Disconnect() end) end

end

    pillBg.BorderSizePixel = 0; pillBg.ZIndex = 32
    mkCorner(pillBg, 10); mkStroke(pillBg, C.pillBorder, 1)
    local dot = Instance.new("Frame", pillBg)
    dot.Size = UDim2.new(0, 14, 0, 14)

    dot.BackgroundColor3 = defaultOn and C.dotOn or C.dotOff

-- Helper: input row (rounded pill row with - / value / + steppers)
local function mInput(label, default, onChange)
    local outer = Instance.new("Frame", menuScroll)
    outer.Size = UDim2.new(1, -10, 0, 44)
    outer.BackgroundTransparency = 1; outer.BorderSizePixel = 0; outer.LayoutOrder = MLO()
    local row = Instance.new("Frame", outer)
    row.Size = UDim2.new(1, -10, 1, -6); row.Position = UDim2.new(0, 5, 0, 3)
    row.BackgroundColor3 = C.rowBg; row.BackgroundTransparency = 0.05
    row.BorderSizePixel = 0; row.ZIndex = 31
    mkCorner(row, 18)
    addSilverStrobe(mkStroke(row, C.silver, 1.5))
    local lbl = Instance.new("TextLabel", row)
    lbl.Size = UDim2.new(1, -150, 1, 0); lbl.Position = UDim2.new(0, 16, 0, 0)
    lbl.BackgroundTransparency = 1; lbl.Text = label
    lbl.TextColor3 = C.rowLabel; lbl.Font = Enum.Font.GothamBold; lbl.TextSize = 12
    lbl.TextXAlignment = Enum.TextXAlignment.Left; lbl.ZIndex = 32
    -- Minus button
    local minus = Instance.new("TextButton", row)
    minus.Size = UDim2.new(0, 22, 0, 22); minus.Position = UDim2.new(1, -130, 0.5, -11)
    minus.BackgroundColor3 = C.inputBg; minus.BorderSizePixel = 0
    minus.Text = "-"; minus.TextColor3 = C.silver; minus.Font = Enum.Font.GothamBlack
    minus.TextSize = 16; minus.ZIndex = 33
    mkCorner(minus, 11); addSilverStrobe(mkStroke(minus, C.silver, 1))
    -- Value box
    local boxWrap = Instance.new("Frame", row)
    boxWrap.Size = UDim2.new(0, 60, 0, 22); boxWrap.Position = UDim2.new(1, -102, 0.5, -11)
    boxWrap.BackgroundColor3 = C.inputBg; boxWrap.BorderSizePixel = 0; boxWrap.ZIndex = 32
    mkCorner(boxWrap, 11)
    local bs = mkStroke(boxWrap, C.silver, 1); addSilverStrobe(bs)
    local box = Instance.new("TextBox", boxWrap)
    box.Size = UDim2.new(1, -6, 1, 0); box.Position = UDim2.new(0, 3, 0, 0)
    box.BackgroundTransparency = 1; box.Text = tostring(default)
    box.TextColor3 = C.inputTxt; box.Font = Enum.Font.GothamBold
    box.TextSize = 12; box.ClearTextOnFocus = false; box.ZIndex = 33
    box.TextXAlignment = Enum.TextXAlignment.Center
    -- Plus button
    local plus = Instance.new("TextButton", row)
    plus.Size = UDim2.new(0, 22, 0, 22); plus.Position = UDim2.new(1, -34, 0.5, -11)
    plus.BackgroundColor3 = C.inputBg; plus.BorderSizePixel = 0
    plus.Text = "+"; plus.TextColor3 = C.silver; plus.Font = Enum.Font.GothamBlack
    plus.TextSize = 16; plus.ZIndex = 33
    mkCorner(plus, 11); addSilverStrobe(mkStroke(plus, C.silver, 1))
    box.Focused:Connect(function() TweenService:Create(bs, TweenInfo.new(0.15), {Color = C.inputFocus}):Play() end)
    box.FocusLost:Connect(function()
        if onChange then
            local n = tonumber(box.Text)
            if n then onChange(n) else box.Text = tostring(default) end
        end
    end)
    local function bump(delta)
        local n = tonumber(box.Text) or default
        local step = (math.abs(n) >= 10) and 1 or 0.1
        n = n + delta * step
        n = math.floor(n * 10 + 0.5) / 10
        box.Text = tostring(n)
        if onChange then pcall(onChange, n) end
    end
    minus.MouseButton1Click:Connect(function() bump(-1) end)
    plus.MouseButton1Click:Connect(function() bump(1) end)
    return box
end

        currentFOV = f
        local t = (f-60)/140
        fill.Size = UDim2.new(t, 0, 1, 0)

        end
    end)

            dragging = false
        end
    end)
    RunService.RenderStepped:Connect(function()
        local c = workspace.CurrentCamera
        if c and c.FieldOfView ~= currentFOV then
            pcall(function() c.FieldOfView = currentFOV end)
        end
    end)

            local c = LP.Character
            local root = c and c:FindFirstChild("HumanoidRootPart")
            if root then
                local v = root.AssemblyLinearVelocity
                if v.Magnitude > 250 then
                    pcall(function() root.AssemblyLinearVelocity = v.Unit * 60 end)
                end
                local av = root.AssemblyAngularVelocity
                if av.Magnitude > 50 then
                    pcall(function() root.AssemblyAngularVelocity = Vector3.zero end)
                end
            end
        end
    end
end)

local KeyBindMap = {
    taunt    = function() if State.tauntActive then stopTaunt(bigBtns.tauntBtn) else startTaunt(bigBtns.tauntBtn) end end,
    drop     = function() if not State.dropEnabled then runDropBrainrot() end end,

    autoR    = function() startAutoPlay("R") end,
    aimbot   = function() if State.aimbotActive then stopAimbot() else startAimbot() end end,
    tpDown   = function() doTpDown() end,
    float_   = function() end,  -- carry speed removed
    floatUp  = function() if _floatActive then stopFloat() else startFloat() end end,
}

-- Keybinds are configured in the menu (Keybinds tab). Input handler below fires them.
-- _rebinding declared above

local function fireKeybind(action)
    if KeyBindMap[action] then pcall(KeyBindMap[action]) end
end

UIS.InputBegan:Connect(function(input, processed)
    local ut = input.UserInputType
    local isKB = ut == Enum.UserInputType.Keyboard
    local isGP = ut == Enum.UserInputType.Gamepad1 or ut == Enum.UserInputType.Gamepad2
        or ut == Enum.UserInputType.Gamepad3 or ut == Enum.UserInputType.Gamepad4
    if not isKB and not isGP then return end
    local keyName = input.KeyCode.Name
    if _rebinding then
        if input.KeyCode == Enum.KeyCode.Escape then
            _rebinding = nil; return
        end
        Cfg.keybinds = Cfg.keybinds or {}
        if isGP then
            -- store gamepad binds separately
            Cfg.gpBinds = Cfg.gpBinds or {}
            Cfg.gpBinds[_rebinding] = keyName
        else
            Cfg.keybinds[_rebinding] = keyName
        end
        pcall(saveConfig)
        _rebinding = nil
        return
    end
    if processed and isKB then return end
    Cfg.keybinds = Cfg.keybinds or {}
    Cfg.gpBinds  = Cfg.gpBinds  or {}
    for action, bound in pairs(Cfg.keybinds) do
        if bound == keyName and KeyBindMap[action] then fireKeybind(action); break end
    end
    if isGP then
        for action, bound in pairs(Cfg.gpBinds) do
            if bound == keyName and KeyBindMap[action] then fireKeybind(action); break end
        end
    end
end)

-- ================================================================
-- NOTIX HUB USER DETECTION v3 (cloud-relay, premium purple/black flash)
-- ----------------------------------------------------------------
-- Cross-client peer detection via Lovable Cloud relay.
-- For confirmed Notix Hub users:
--   * Strips the regular ESP (box + name billboard) repeatedly
--   * Adds a premium "Notix hub user" title above their head (raised)
--   * Title flashes silver <-> black with glow
--   * Highlight pulses silver/black
-- For everyone else: regular ESP from the hub stays as-is.
-- ================================================================
task.spawn(function()
    local ok_outer, err_outer = pcall(function()
        local Players      = game:GetService("Players")
        local HttpService  = game:GetService("HttpService")
        local RunService   = game:GetService("RunService")
        local LocalPlayer  = Players.LocalPlayer

        -- ---- Config ----
        local RELAY_BASE   = "https://project--1ae4b19d-1259-42e6-a26f-051761ea8a31.lovable.app"
        local REPORT_URL   = RELAY_BASE .. "/api/public/notixhub/report"
        local PEERS_URL    = RELAY_BASE .. "/api/public/notixhub/peers"

        local REPORT_EVERY = 15
        local POLL_EVERY   = 6
        local FRESH_WINDOW = 60

        local TAG_NAME     = "NotixHubUserTag_v3"
        local HL_NAME      = "NotixHubUserHighlight_v3"
        local FRAME_NAME   = "NotixHubUserFrame_v3"

        -- ESP marker names used by the hub itself — stripped from Notix Hub users.
        local STRIP_NAMES = {
            "CashHubHitbox", "CashHubName",
            "CashHubESP", "CashHubESP_Name",
        }

        -- ---- HTTP shim across executors ----
        local function httpRequest(opts)
            local req = (syn and syn.request)
                or (http and http.request)
                or (fluxus and fluxus.request)
                or http_request
                or request
            if not req then return nil end
            local ok, res = pcall(req, opts)
            if ok then return res end
            return nil
        end

        local function getJobId()
            local id = game.JobId
            if id == nil or id == "" then
                return "studio-" .. tostring(game.PlaceId)
            end
            return id
        end

        -- ---- Beacon ----
        local function sendBeacon()
            local body = HttpService:JSONEncode({
                userId   = tostring(LocalPlayer.UserId),
                jobId    = getJobId(),
                username = LocalPlayer.Name,
            })
            httpRequest({
                Url     = REPORT_URL,
                Method  = "POST",
                Headers = { ["Content-Type"] = "application/json" },
                Body    = body,
            })
        end

        -- ---- Peer cache ----
        local notixhubUserIds = {} -- [userIdString] = true

        local function fetchPeers()
            local res = httpRequest({
                Url    = PEERS_URL .. "?jobId=" .. HttpService:UrlEncode(getJobId()),
                Method = "GET",
            })
            if not res or not res.Body then return end
            local ok, data = pcall(function() return HttpService:JSONDecode(res.Body) end)
            if not ok or type(data) ~= "table" or not data.users then return end

            local fresh = {}
            local now = data.now or os.time()
            for _, u in ipairs(data.users) do
                if u.userId and (now - (u.lastSeen or 0)) <= FRESH_WINDOW then
                    if tostring(u.userId) ~= tostring(LocalPlayer.UserId) then
                        fresh[tostring(u.userId)] = true
                    end
                end
            end
            notixhubUserIds = fresh
        end

        -- Local-machine peer flag (same executor)
        pcall(function()
            if getgenv then
                getgenv().JHubUser = true
                getgenv().JHubPeers = getgenv().JHubPeers or {}
                getgenv().JHubPeers[LocalPlayer.Name] = true
            end
        end)

        local function isJHubUser(plr)
            if plr == LocalPlayer then return false end
            if notixhubUserIds[tostring(plr.UserId)] then return true end
            if getgenv and getgenv().JHubPeers and getgenv().JHubPeers[plr.Name] then
                return true
            end
            return false
        end

        -- ---- Strip the hub's normal ESP from a player's char ----
        local function stripNormalESP(char)
            if not char then return end
            for _, n in ipairs(STRIP_NAMES) do
                local x = char:FindFirstChild(n)
                if x then pcall(function() x:Destroy() end) end
            end
        end

        -- ---- Build premium visuals ----
        local function buildVisuals(plr)
            local char = plr.Character
            if not char then return end
            local head = char:FindFirstChild("Head")
            if not head then return end

            -- Highlight (silver/black pulse target)
            if not char:FindFirstChild(HL_NAME) then
                local hl = Instance.new("Highlight")
                hl.Name              = HL_NAME
                hl.Adornee           = char
                hl.FillColor         = Color3.fromRGB(210, 210, 215)
                hl.OutlineColor      = Color3.fromRGB(20, 20, 20)
                hl.FillTransparency  = 0.45
                hl.OutlineTransparency = 0
                hl.DepthMode         = Enum.HighlightDepthMode.AlwaysOnTop
                hl.Parent            = char
            end

            -- Tag billboard (text-only, no frame box, no star, slightly smaller)
            if not head:FindFirstChild(TAG_NAME) then
                local bb = Instance.new("BillboardGui")
                bb.Name           = TAG_NAME
                bb.Adornee        = head
                bb.Size           = UDim2.new(0, 190, 0, 36)  -- slightly smaller
                bb.StudsOffset    = Vector3.new(0, 4.2, 0)    -- HIGHER than normal ESP
                bb.AlwaysOnTop    = true
                bb.LightInfluence = 0
                bb.MaxDistance    = 1500
                bb.Parent         = head

                -- Just a transparent container so the flash loop has a stable parent
                local frame = Instance.new("Frame")
                frame.Name                   = FRAME_NAME
                frame.BackgroundTransparency = 1
                frame.BorderSizePixel        = 0
                frame.Size                   = UDim2.new(1, 0, 1, 0)
                frame.Parent                 = bb

                -- Main label (text only)
                local label = Instance.new("TextLabel", frame)
                label.Name        = "Label"
                label.BackgroundTransparency = 1
                label.Position    = UDim2.new(0, 0, 0, 0)
                label.Size        = UDim2.new(1, 0, 1, 0)
                label.Font        = Enum.Font.GothamBold
                label.TextScaled  = true
                label.Text        = "Notix Hub user"
                label.TextColor3  = Color3.fromRGB(235, 235, 240)
                label.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
                label.TextStrokeTransparency = 0.15
                label.TextXAlignment = Enum.TextXAlignment.Center

                -- Text gradient (silver/black flash target)
                local tgrad = Instance.new("UIGradient", label)
                tgrad.Name = "TextFlashGradient"
                tgrad.Rotation = 0
                tgrad.Color = ColorSequence.new({
                    ColorSequenceKeypoint.new(0,    Color3.fromRGB(255, 255, 255)),
                    ColorSequenceKeypoint.new(0.5,  Color3.fromRGB(160, 70, 220)),
                    ColorSequenceKeypoint.new(1,    Color3.fromRGB(20,  20,  25)),
                })
            end
        end

        local function clearVisuals(plr)
            local char = plr.Character
            if not char then return end
            local hl = char:FindFirstChild(HL_NAME)
            if hl then hl:Destroy() end
            local head = char:FindFirstChild("Head")
            if head then
                local bb = head:FindFirstChild(TAG_NAME)
                if bb then bb:Destroy() end
            end
        end

        -- ---- Render-stepped flash + ESP strip loop ----
        RunService.RenderStepped:Connect(function()
            local t   = tick() * 0.9
            local mix = (math.sin(t) + 1) * 0.5  -- 0..1
            -- Silver <-> Black mix
            local v   = math.floor(20 + mix * 215)        -- 20..235
            local col = Color3.fromRGB(v, v, math.min(255, v + 8))
            local fillTrans = 0.20 + (1 - mix) * 0.40     -- 0.20..0.60
            local strokeTrans = 0.05 + (1 - mix) * 0.30

            for _, plr in ipairs(Players:GetPlayers()) do
                if isJHubUser(plr) then
                    local char = plr.Character
                    if char then
                        -- Always strip the normal ESP markers (handles re-creation by toggleESP)
                        stripNormalESP(char)
                        buildVisuals(plr)

                        local hl = char:FindFirstChild(HL_NAME)
                        if hl then
                            hl.FillColor         = col
                            hl.OutlineColor      = Color3.fromRGB(255 - v, 255 - v, 255 - v)
                            hl.FillTransparency  = fillTrans
                        end

                        local head = char:FindFirstChild("Head")
                        if head then
                            local bb = head:FindFirstChild(TAG_NAME)
                            if bb then
                                local frame = bb:FindFirstChild(FRAME_NAME)
                                if frame then
                                    local stroke = frame:FindFirstChild("PremiumStroke")
                                    if stroke then
                                        stroke.Color        = col
                                        stroke.Transparency = strokeTrans
                                    end
                                    local grad = frame:FindFirstChild("PremiumGradient")
                                    if grad then
                                        grad.Rotation = (tick() * 30) % 360
                                    end
                                    local label = frame:FindFirstChild("Label")
                                    if label then
                                        local tgrad = label:FindFirstChild("TextFlashGradient")
                                        if tgrad then
                                            -- shift the silver/black gradient across the text
                                            local shift = (math.sin(tick() * 1.6) + 1) * 0.5
                                            tgrad.Offset = Vector2.new(shift - 0.5, 0)
                                        end
                                        label.TextColor3 = col
                                    end
                                    local crown = frame:FindFirstChild("Crown")
                                    if crown then
                                        crown.TextColor3 = col
                                    end
                                end
                            end
                        end
                    end
                else
                    -- Not a Notix Hub user: make sure our premium tag isn't lingering.
                    clearVisuals(plr)
                end
            end
        end)

        -- ---- Beacon + poll loops ----
        task.spawn(function()
            while true do
                pcall(sendBeacon)
                task.wait(REPORT_EVERY)
            end
        end)

        task.spawn(function()
            while true do
                pcall(fetchPeers)
                task.wait(POLL_EVERY)
            end
        end)

        -- Re-beacon on respawn
        LocalPlayer.CharacterAdded:Connect(function()
            task.wait(0.5)
            pcall(sendBeacon)
        end)

        -- Also strip ESP aggressively whenever a Notix Hub user respawns.
        local function hookPlayer(plr)
            plr.CharacterAdded:Connect(function(char)
                task.wait(0.6)
                if isJHubUser(plr) then
                    stripNormalESP(char)
                    buildVisuals(plr)
                end
            end)
        end
        for _, p in ipairs(Players:GetPlayers()) do hookPlayer(p) end
        Players.PlayerAdded:Connect(hookPlayer)
        Players.PlayerRemoving:Connect(function(plr)
            notixhubUserIds[tostring(plr.UserId)] = nil
        end)
