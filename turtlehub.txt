function ACBypass()
    local function checkAndBlock(remote, args)
        local foundStop = false
        for _, v in pairs(args) do
            if type(v) == "string" and string.find(string.lower(v), "stop") then
                foundStop = true
                break
            end
        end

        if foundStop then
            return true
        end
        return false
    end

    local ok = pcall(function()
        local oldRemote
        oldRemote = hookfunction(Instance.new('RemoteEvent').FireServer, newcclosure(function(self, ...)
            local args = {...}
            if checkAndBlock(self, args) then
                return nil
            end
            return oldRemote(self, ...)
        end))
    end)

    if not ok then
        cloneref(game.Players).LocalPlayer:Kick("not supported executor!")
    end
end

local SAVEFILE = "TH_save.json"
local S        = { toggles = {}, panels = {} }

local function _writeData()
    local ok, enc = pcall(HS.JSONEncode, HS, { toggles = S.toggles, panels = S.panels })
    if ok then pcall(writefile, SAVEFILE, enc) end
end

local function _readData()
    local ok, raw = pcall(readfile, SAVEFILE)
    if ok and type(raw) == "string" and #raw > 2 then
        local ok2, d = pcall(HS.JSONDecode, HS, raw)
        if ok2 and type(d) == "table" then
            if type(d.toggles) == "table" then S.toggles = d.toggles end
            if type(d.panels)  == "table" then S.panels  = d.panels  end
        end
    end
end
_readData()

local _pending = false
local function _flush()
    if _pending then return end
    _pending = true
    task.delay(0.8, function() _pending = false; _writeData() end)
end

local activeDrag  = nil
local DRAG_THRESH = 6

UIS.InputChanged:Connect(function(i)
    if not activeDrag then return end
    if i.UserInputType ~= Enum.UserInputType.MouseMovement and
       i.UserInputType ~= Enum.UserInputType.Touch then return end
    local dx = i.Position.X - activeDrag.startMouseX
    local dy = i.Position.Y - activeDrag.startMouseY
    if not activeDrag.dragging then
        if math.abs(dx) < DRAG_THRESH and math.abs(dy) < DRAG_THRESH then return end
        activeDrag.dragging = true
    end
    local f  = activeDrag.frame
    local nx = activeDrag.startPosX + dx
    local ny = activeDrag.startPosY + dy
    if activeDrag.subPanel then
        local hw = f.AbsoluteSize.X / 2
        local hh = f.AbsoluteSize.Y / 2
        nx = math.clamp(nx, -hw, vp.X - hw)
        ny = math.clamp(ny, -hh, vp.Y - hh)
    else
        nx = math.clamp(nx, 0, vp.X - f.AbsoluteSize.X)
        ny = math.clamp(ny, 0, vp.Y - f.AbsoluteSize.Y)
    end
    f.Position = UDim2.fromOffset(nx, ny)
end)

UIS.InputEnded:Connect(function(i)
    if i.UserInputType ~= Enum.UserInputType.MouseButton1 and
       i.UserInputType ~= Enum.UserInputType.Touch then return end
    if activeDrag then
        if activeDrag.onEnd then activeDrag.onEnd() end
        activeDrag = nil
    end
end)

local function makeDrag(frame, handle, onEnd, subPanel)
    handle = handle or frame
    handle.InputBegan:Connect(function(i)
        if i.UserInputType ~= Enum.UserInputType.MouseButton1 and
           i.UserInputType ~= Enum.UserInputType.Touch then return end
        local p = frame.Position
        activeDrag = {
            frame       = frame,
            startPosX   = p.X.Offset,
            startPosY   = p.Y.Offset,
            startMouseX = i.Position.X,
            startMouseY = i.Position.Y,
            dragging    = false,
            onEnd       = onEnd,
            subPanel    = subPanel,
        }
    end)
end

    local numBtns = {}

    local function buildPad()

        numBtns = {}

        local bw  = math.floor((PW_H - pad*2 - gap*2) / 3)

            local b = Instance.new("TextButton")
            b.Text                  = lbl
            b.Size                  = UDim2.fromOffset(bw, bh)
            b.Position              = UDim2.fromOffset(pad + (i-1)*(bw+gap), pad)
            b.BackgroundColor3      = rgb(14,50,28)
            b.BackgroundTransparency = 0.05
            b.TextColor3            = C.White
            b.TextSize              = mobile and 20 or 24
            b.Font                  = FONT_B
            b.BorderSizePixel       = 0
            b.ZIndex                = 27
            b.AutoButtonColor       = false
            b.Parent                = numCon
            corner(b, 10); stroke(b, C.Border, 0.2, 1.5)

                Tween:Create(b, TweenInfo.new(0.1), {BackgroundColor3=rgb(0,80,38), BackgroundTransparency=0, TextColor3=C.GreenB}):Play()

function equipTool(toolName)
    local t = player.Backpack:FindFirstChild(toolName) or getChar():FindFirstChild(toolName)
    if t then local hum = getHumanoid() hum:EquipTool(t); return t end
    for _,v in pairs(player.Backpack:GetChildren()) do
        if v:IsA("Tool") and v.Name:lower():find(toolName:lower()) then local hum = getHumanoid() hum:EquipTool(v); return v end
    end
end

function getNearestTarget()
    local hrp = getHrp()
    if not hrp then return end
    local closestTarget = nil
    local closestDistance = math.huge
    for _, other in pairs(game.Players:GetPlayers()) do
        if other ~= player then
            local distance = (hrp.Position - other.Character.PrimaryPart.Position).Magnitude
            if distance < closestDistance then
                closestDistance = distance
                closestTarget = other
            end
        end
    end
    return closestTarget
end

function getRealAPRemote()
    local children = Packages.Net:GetChildren()
    local byIdx    = {}
    local byName   = {}
    for i, obj in ipairs(children) do
        byIdx[i]          = obj
        byName[obj.Name]  = i
    end
    local anchorIdx = byName["RF/a0e78691-cb9b-4efc-ac08-9c06fea70059"]
    if anchorIdx then
        local actual = byIdx[anchorIdx + 1]
        if actual then return actual end
    end
    return nil
end

function checkCommandCD(command)
    local commandButton = ScrollingFrame:FindFirstChild(command)
    if commandButton then
        local Timer = commandButton:FindFirstChild("Timer")
        if Timer and Timer.Visible then return false end
    elseif not commandButton then
        return false
    end
    for name, data in pairs(AdminCommands) do
        if name == command then
            if (workspace:GetServerTimeNow() - (State.CommandLastUsed[command] or 0)) < data.cooldown then
                return false
            end
        end
    end
    return true
end

function fireAdmin(...)
    if not Data.RealAPRemote then return nil end
    local args = {...}
    if not checkCommandCD(args[3]) then
        Library:Notify("Turtle Hub Notify", tostring(args[3]) .. " has cooldown yet!", 2)
        return nil
    end
    task.spawn(function()
        State.CommandLastUsed[args[3]] = workspace:GetServerTimeNow()
        if args[3] == "balloon" and args[2] and args[2].Character then
            State.BalloonedPlayers[args[2].Character] = true
            task.delay(15, function()
                if args[2].Character then State.BalloonedPlayers[args[2].Character] = nil end
            end)
        end
        Data.RealAPRemote:InvokeServer(unpack(args))
    end)
    return true
end

end)

function findTargetBase()
    local myBase = Data.MY_BASE.Model
    if not myBase or not myBase.PrimaryPart then return end
    local shortestDistance = math.huge
    local closestModel = nil
    for _, model in ipairs(plots:GetChildren()) do
        if model:IsA("Model") and model ~= myBase and model.PrimaryPart then
            local isOtherOwner = false
            for _, obj in ipairs(model:GetDescendants()) do
                if obj:IsA("TextLabel") then
                    local txt = string.lower(obj.Text)
                    if txt ~= "" and not string.find(txt, "empty") then
                        isOtherOwner = true; break
                    end
                end
            end
            if isOtherOwner then
                local distance = (Data.CRITERIA_POS - model.PrimaryPart.Position).Magnitude
                if distance < shortestDistance then
                    shortestDistance = distance; closestModel = model
                end
            end
        end
    end
    if closestModel then
        if closestModel.PrimaryPart and Data.MY_BASE.Model and Data.MY_BASE.Model.PrimaryPart then
            if (closestModel.PrimaryPart.Position - Data.REFERNCE_POS).Magnitude < (Data.MY_BASE.Model.PrimaryPart.Position - Data.REFERNCE_POS).Magnitude then
                return closestModel, "2"
            else
                return closestModel, "1"
            end
        end
    end
end

function buildStealCallbacks(prompt)
    if Data.promptInfo[prompt] then return end
    local data = { holdCallbacks = {}, triggerCallbacks = {}, ready = true }
    local ok1, conns1 = pcall(getconnections, prompt.PromptButtonHoldBegan)
    if ok1 then for _, c in ipairs(conns1) do table.insert(data.holdCallbacks, c.Function) end end
    local ok2, conns2 = pcall(getconnections, prompt.Triggered)
    if ok2 then for _, c in ipairs(conns2) do table.insert(data.triggerCallbacks, c.Function) end end
    Data.promptInfo[prompt] = data
end

function executeInternalStealAsync(prompt)
    local data = Data.promptInfo[prompt]
    if not data or not data.ready or State.teleporting then return end
    data.ready = false
    State.teleporting = true
    local StealProgress = 0
    local tpDone = false

    task.spawn(function()
        for _, fn in ipairs(data.holdCallbacks) do task.spawn(fn) end
        local startTime = tick()
        while tick() - startTime < 1.5 do
            StealProgress = (tick() - startTime) / 1.5
            if StealProgress >= 0.73 and not tpDone then
                tpDone = true
                carpetTP()
            end
            task.wait()
        end
        StealProgress = 1
        for _, fn in ipairs(data.triggerCallbacks) do task.spawn(fn) end
        fastTeleport(Data.Waypoints[tonumber(Data.TARGET_BASE.Case)]["SEMI_POS"])

        if ScriptSettings.USE_POTION then
            task.spawn(function()
                local giantPotion = equipTool("Giant")
                giantPotion:Activate()
                giantPotion.Parent = player.Backpack
            end)
            equipTool("Carpet")
        end

        local target = getNearestTarget()

        if ScriptSettings.AUTO_AP_SPAM then
            task.spawn(function()
                for _, command in ipairs(Data.AP_SPAM_COMMANDS) do
                    fireAdmin(Data.AP_EXECUTE_KEY, target, command)
                    task.wait()
                end
            end)
        end

        if Data.MY_BASE.Model then
            local cashPad = Data.MY_BASE.Model:WaitForChild("CashPad")
            local basePart = cashPad:WaitForChild("structure base home")
            task.wait(0.5)
            local hrp = getHrp()
            if hrp and State.stealing and ScriptSettings.AUTO_TWEEN_TO_BASE then
                Library:Notify("Turtle Hub Notify", "Steal successful! Tweening to base...", 2)

                -- レイキャストで安全な着地点Y座標を求める
                local rawPos = basePart.Position
                local targetXZ = Vector3.new(rawPos.X, 0, rawPos.Z)

                -- Humanoid:MoveToで歩かせつつVelocityで速度を制御
                local hum = getHumanoid()
                if hum then
                    _tweening = true  -- applySpeedの干渉をブロック
                    local moveConn
                    moveConn = RunService.Heartbeat:Connect(function()
                        if not State.stealing or not ScriptSettings.AUTO_TWEEN_TO_BASE then
                            _tweening = false
                            moveConn:Disconnect()
                            return
                        end
                        local currentHrp = getHrp()
                        if not currentHrp then
                            _tweening = false
                            moveConn:Disconnect()
                            return
                        end
                        local pos = currentHrp.Position
                        local diffXZ = targetXZ - Vector3.new(pos.X, 0, pos.Z)
                        if diffXZ.Magnitude < 2 then
                            _tweening = false
                            moveConn:Disconnect()
                            return
                        end
                        -- MoveToで方向を指示しVelocityでTween速度を適用
                        hum:MoveTo(Vector3.new(rawPos.X, pos.Y, rawPos.Z))
                        local dir = diffXZ.Unit
                        currentHrp.Velocity = dir * ScriptSettings.TWEEN_SPEED + Vector3.new(0, currentHrp.Velocity.Y, 0)
                    end)
                end
            end
        end

        task.wait(0.2)
        data.ready = true
        State.teleporting = false
        StealProgress = 0
    end)
end

        stealBarFill.Size = UDim2.new(0, barW * progress, 1, -2)
        stealBarFill.BackgroundColor3 = Color3.fromRGB(
            0, math.floor(195 + progress * 50), math.floor(progress * 30))
        if progress >= 1 then
            _stealProgConn:Disconnect()
            _stealProgConn = nil
            stealBarFill.Size = UDim2.new(0, barW, 1, -2)
            stealBarLabel.Text = "STOLEN!"
            stealBarFill.BackgroundColor3 = C.GreenB
            task.delay(0.8, function() stealBarBg.Visible = false end)
        end
    end)
end

local function agv2FindNearestPrompt()
    local hrp = getHrp()
    if not hrp then return nil end
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return nil end
    local nearest, dist = nil, math.huge
    for _, plot in ipairs(plots:GetChildren()) do
        local sign = plot:FindFirstChild("PlotSign")
        if sign then
            local yb = sign:FindFirstChild("YourBase")
            if yb and yb:IsA("BillboardGui") and yb.Enabled then continue end
        end
        local pods = plot:FindFirstChild("AnimalPodiums")
        if not pods then continue end
        for _, pod in ipairs(pods:GetChildren()) do
            local base = pod:FindFirstChild("Base")
            if not base then continue end
            local spawn = base:FindFirstChild("Spawn")
            if not spawn then continue end
            local d = (spawn.Position - hrp.Position).Magnitude
            if d <= 8 and d < dist then
                local att = spawn:FindFirstChild("PromptAttachment")
                if att then
                    for _, p in ipairs(att:GetChildren()) do
                        if p:IsA("ProximityPrompt") and p.ActionText and p.ActionText:find("Steal") then
                            nearest, dist = p, d
                        end
                    end
                end
            end
        end
    end
    return nearest
end

local function agv2ExecuteSteal(prompt)
    if _agv2IsStealing then return end
    if not _agv2StealData[prompt] then
        _agv2StealData[prompt] = {hold = {}, trigger = {}, ready = true}
        if getconnections then
            for _, c in ipairs(getconnections(prompt.PromptButtonHoldBegan)) do
                if c.Function then table.insert(_agv2StealData[prompt].hold, c.Function) end
            end
            for _, c in ipairs(getconnections(prompt.Triggered)) do
                if c.Function then table.insert(_agv2StealData[prompt].trigger, c.Function) end
            end
        end
    end
    local data = _agv2StealData[prompt]
    if not data.ready then return end
    data.ready = false
    _agv2IsStealing = true
    task.spawn(function()
        showStealProgress(0.4)
        for _, f in ipairs(data.hold) do pcall(function() f() end) end
        task.wait(0.4)  -- V2
        for _, f in ipairs(data.trigger) do pcall(function() f() end) end
        if ScriptSettings.USE_POTION then
            task.spawn(function()
                task.wait(0.2)
                local backpack = player:WaitForChild("Backpack")
                local char = player.Character
                local tool = (char and char:FindFirstChild("Giant Potion"))
                    or backpack:FindFirstChild("Giant Potion")
                if tool then
                    local hum = getHumanoid()
                    if hum then hum:EquipTool(tool) end
                    task.wait(0.05)
                    tool:Activate()
                end
            end)
        end
        data.ready = true
        _agv2IsStealing = false
    end)
end

local function v1Stop()
    _v1Active = false
    _v1Prompt = nil
    if _v1Conn then _v1Conn:Disconnect(); _v1Conn = nil end

local function v1FindPrompt()
    local hrp = getHrp()
    if not hrp then return nil end
    if not Data.TARGET_BASE.Model then return nil end
    local pods = Data.TARGET_BASE.Model:FindFirstChild("AnimalPodiums")
    if not pods then return nil end
    local nearest, dist = nil, math.huge
    for _, desc in ipairs(pods:GetDescendants()) do
        if desc:IsA("ProximityPrompt") and desc.Enabled and desc.ActionText == "Steal" then
            local part = desc.Parent
            if part and part:IsA("BasePart") then
                local d = (part.Position - hrp.Position).Magnitude
                if d < dist then dist = d; nearest = desc end
            end
        end
    end
    return nearest
end

local function v1ExecuteOnce(prompt)
    if _v1Active then return end
    _v1Active = true
    _v1Prompt = prompt

    local holdFns, trigFns = {}, {}
    local ok1, c1 = pcall(getconnections, prompt.PromptButtonHoldBegan)
    if ok1 then for _, c in ipairs(c1) do if c.Function then table.insert(holdFns, c.Function) end end end
    local ok2, c2 = pcall(getconnections, prompt.Triggered)
    if ok2 then for _, c in ipairs(c2) do if c.Function then table.insert(trigFns, c.Function) end end end

    task.spawn(function()
        for _, f in ipairs(holdFns) do pcall(f) end
        task.wait(0.6)
        if prompt and prompt.Enabled and _v1Active then
            for _, f in ipairs(trigFns) do pcall(f) end
        end
        _v1Active = false
        _v1Prompt = nil
    end)
end

local function applySpeed()
    if _tweening then return end  -- Tween中はスキップ
    local hrp = getHrp()
    local hum = getHumanoid()
    if hrp and hum then
        if hum.WalkSpeed > 25 then
            State.stealing = false
            if ScriptSettings.NORMAL_BOOST and hum.MoveDirection.Magnitude > 0 then
                hrp.Velocity = hum.MoveDirection * ScriptSettings.NORMAL_SPEED + Vector3.new(0, hrp.Velocity.Y, 0)
            end
        else
            State.stealing = true
            if ScriptSettings.STEAL_BOOST and hum.MoveDirection.Magnitude > 0 then
                hrp.Velocity = hum.MoveDirection * ScriptSettings.STEAL_SPEED + Vector3.new(0, hrp.Velocity.Y, 0)
            end
        end
    end
end

function fpsBoost()
    local decalsyeeted = true
    local g = game; local w = g.Workspace; local l = g.Lighting; local t = w.Terrain
    sethiddenproperty(l, "Technology", 2)
    sethiddenproperty(t, "Decoration", false)
    t.WaterWaveSize = 0; t.WaterWaveSpeed = 0; t.WaterReflectance = 0; t.WaterTransparency = 0
    l.GlobalShadows = false; l.FogEnd = 9e9; l.Brightness = 0
    settings().Rendering.QualityLevel = "Level01"
    local function optimizePart(v)
        if v:IsA("Part") or v:IsA("Union") or v:IsA("CornerWedgePart") or v:IsA("TrussPart") then
            v.Material = "Plastic"; v.Reflectance = 0
        elseif v:IsA("Decal") or v:IsA("Texture") and decalsyeeted then
            v.Transparency = 1
        elseif v:IsA("ParticleEmitter") or v:IsA("Trail") then
            v.Lifetime = NumberRange.new(0)
        elseif v:IsA("Explosion") then
            v.BlastPressure = 1; v.BlastRadius = 1
        elseif v:IsA("Fire") or v:IsA("SpotLight") or v:IsA("Smoke") or v:IsA("Sparkles") then
            v.Enabled = false
        elseif v:IsA("MeshPart") then
            v.Material = "Plastic"; v.Reflectance = 0; v.TextureID = 10385902758728957
        end
    end
    for _, v in pairs(w:GetDescendants()) do
        if v:IsA("Part") or v:IsA("Union") or v:IsA("CornerWedgePart") or v:IsA("TrussPart") or v:IsA("MeshPart") then
            optimizePart(v)
        end
    end
    for _, e in pairs(l:GetChildren()) do
        if e:IsA("BlurEffect") or e:IsA("SunRaysEffect") or e:IsA("ColorCorrectionEffect") or e:IsA("BloomEffect") or e:IsA("DepthOfFieldEffect") then
            e.Enabled = false
        end
    end
end

local _arConns = {}

local function startAntiRagdoll()
    if _arConns.antiRagdoll then return end

    _arConns.antiRagdoll = RunService.Heartbeat:Connect(function()
        if not ScriptSettings.ANTI_RAGDOLL then return end

        local char = player.Character
        if not char then return end

        local hum  = char:FindFirstChildOfClass("Humanoid")
        local root = char:FindFirstChild("HumanoidRootPart")
        if not (hum and root) then return end

        local s = hum:GetState()
        local ragdolled = (s == Enum.HumanoidStateType.Physics or s == Enum.HumanoidStateType.Ragdoll or s == Enum.HumanoidStateType.FallingDown)

        local endTime = player:GetAttribute("RagdollEndTime")
        if endTime and (endTime - workspace:GetServerTimeNow()) > 0 then
            ragdolled = true
        end

        if ragdolled then
            pcall(function() player:SetAttribute("RagdollEndTime", workspace:GetServerTimeNow()) end)

            for _, d in ipairs(char:GetDescendants()) do
                if d:IsA("BallSocketConstraint") or (d:IsA("Attachment") and d.Name:find("RagdollAttachment")) then
                    d:Destroy()
                end
            end

            for _, obj in ipairs(char:GetDescendants()) do
                if obj:IsA("Motor6D") and obj.Enabled == false then
                    obj.Enabled = true
                end
            end

            if hum.Health > 0 then
                hum:ChangeState(Enum.HumanoidStateType.Running)
            end

            workspace.CurrentCamera.CameraSubject = hum
            root.Anchored = false
            root.AssemblyLinearVelocity = Vector3.zero
            root.AssemblyAngularVelocity = Vector3.zero
        end
    end)
end

local function stopAntiRagdoll()
    if _arConns.antiRagdoll then
        _arConns.antiRagdoll:Disconnect()
        _arConns.antiRagdoll = nil
    end
end

            local now = tick()
            if now - State.lastJump >= 0 then
                State.lastJump = now
                hrp.Velocity = Vector3.new(hrp.Velocity.X, 50, hrp.Velocity.Z)
            end
        end
    end
end
